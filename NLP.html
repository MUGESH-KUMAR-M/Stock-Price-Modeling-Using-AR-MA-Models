from itertools import product
import warnings
warnings.filterwarnings('ignore')

p_range = range(0, 4)
q_range = range(0, 4)
results = []

print("Grid Search for Optimal ARIMA(p,d,q) Parameters...")
for p, q in product(p_range, q_range):
    try:
        model = ARIMA(diff_price, order=(p, 0, q))
        result = model.fit()
        results.append({
            'p': p, 'q': q,
            'AIC': result.aic,
            'BIC': result.bic
        })
    except:
        pass

results_df = pd.DataFrame(results).sort_values('AIC')
print("\nTop 5 Models by AIC:")
print(results_df.head())

best_p, best_q = results_df.iloc[0]['p'], results_df.iloc[0]['q']
best_model = ARIMA(diff_price, order=(int(best_p), 0, int(best_q)))
best_result = best_model.fit()
print(f"\nBest Model: ARIMA({int(best_p)},0,{int(best_q)})")

def calculate_metrics(model_result, model_name):
    return {
        'Model': model_name,
        'AIC': model_result.aic,
        'BIC': model_result.bic,
        'Log-Likelihood': model_result.llf
    }

    calculate_metrics(ar_result, 'AR(2)'),
